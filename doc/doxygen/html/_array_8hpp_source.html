162  virtual size_t NewSize(size_t size) const {
163  size_t count = 0;
164  // Lets increase the buffer size by default buffer
165  // sized chunks. Note the desired new size is always
166  // needed size + 1. The size in chunks is calculated
167  // as (new size + (chunk size - 1)) / chunk size.
168  // since new size = needed size + 1 then we have
169  // chunks = (needed size + chunk size) / chunk size.
170  // Finally we need bytes which is chunks * chunk size
171  // which can be reduced to
172  // ((needed size / chunk size) + 1) * chunk size
173  //
174  count = (size = ((size/VDefaultSize)+1)*VDefaultSize);
175  return count;
176  }
